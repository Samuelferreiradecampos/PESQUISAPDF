import fitz # PyMuPDF import re import os def extrair_nomes_numeros(pdf_path): nomes = [] numeros = [] doc = fitz.open(pdf_path) for page_num in range(doc.page_count): page = doc[page_num] text = page.get_text("text") # Usar regex para encontrar nomes e números padrao_nome = re.compile(r'\bEMEF PROFA MARIA APARECIDA DO NASCIMENTO\b', re.IGNORECASE) padrao_numero = re.compile(r'\b\d{7}\b') nomes.extend(padrao_nome.findall(text)) numeros.extend(padrao_numero.findall(text)) doc.close() return nomes, numeros if __name__ == "__main__": # Caminho do PDF no Desktop pdf_path_no_desktop = os.path.join(os.path.expanduser('~'), 'Desktop', 'DIARIO OFICIAL PDF.pdf') # Exemplo de uso nomes_encontrados, numeros_encontrados = extrair_nomes_numeros(pdf_path_no_desktop) # Exibir resultados apenas se houver algo para mostrar if nomes_encontrados: print("Nomes encontrados:") for nome in nomes_encontrados: print(nome) else: print("Nenhum nome encontrado.") if numeros_encontrados: print("Números encontrados:") for numero in numeros_encontrados: print(numero) else: print("Nenhum número encontrado.")
